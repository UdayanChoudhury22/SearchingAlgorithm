class Solution {
    int binarysearch(int arr[], int n, int key) {
        // code here
        int high = arr.length-1;
        int low = 0;
        while(low<=high)
        {
            int mid = low+(high-low)/2;
            if(arr[mid] == key)
                return mid;
            else if(arr[mid] < key)
                low = mid+1;
            else
                high = mid-1;
        }
        return -1;
    }
}
